fruits = ["Banana", "Apple", "Grape", "Orange", "Mango"]

print("Original:", fruits) # show original

abc = sorted(fruits)
print("sorted() :", abc) # Show in alphabetical order

print("Yet original:", fruits) # show yet original

fruits.sort()
print("after sort():", fruits) # show again the original list
